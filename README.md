比如 1000 / 2000 = 0.5 也就是容器的尺寸是图片的0.5倍

var getCompressSize = function(maxW, maxH, w, h){
    var scale = this.scale = Math.min(maxW / w, maxH / h, 1);
    return {
        width: Math.round(w * scale),
        height: Math.round(h * scale),
    };
};
